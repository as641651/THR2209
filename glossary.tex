\newglossaryentry{basis}{%
  name={basis},
  description={A set of \gls{linearly independent} vectors $B$ are said to form a basis of a vector space $V$, if every vector in $V$ can be represented as a \gls{linear combination} of $B$}
}

\newglossaryentry{linearly independent}{%
  name={linearly independent},
  description={A set of vectors $S$ are said to be linearly independent if every vector in the set \textbf{cannot} be represented as a \gls{linear combination} of other vectors in the set $S$}
}

\newglossaryentry{linear combination}{%
  name={linear combination},
  description={A linear combination of set of vectors $\{\textbf{v}_{1}, \textbf{v}_{2},...,\textbf{v}_{n}\}$ is defined as $a_{1}\textbf{v}_{1} + a_{2}\textbf{v}_{2} + ... + a_{n}\textbf{v}_{n}$, where $a_{1}, a_{2}, .., a_{n}$ are some scalar coefficients which represent the co-ordinates relative to the basis $B$}
}
